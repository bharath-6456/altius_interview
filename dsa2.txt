#include<bits/stdc++.h>
using namespace std;
int main(){
    int n,k;
  cin>>n>>k;
  int arr[n];
  for(int i=0;i<n;i++){
    cin>>arr[i];
  }
  vector<int>ans;
  for(int i=0;i<n-k+1;i++)
    {
        vector<int>temp;
        for(int j=i;j<i+k;j++)
        {
            temp.push_back(arr[j]);
        }
        sort(temp.begin(),temp.end());
        ans.push_back(temp[k/2]);
        temp.clear();

        }
    
    for(int i=0;i<ans.size();i++)
    {
        cout<<ans[i]<<" ";
    }
   
}
